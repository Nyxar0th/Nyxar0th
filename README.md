## code_showcase
<pre>
// nyx.engine
class System {
  constructor(config) {
    this.mode = config.mode ?? "silent";
    this.purple = config.purple ?? true;
  }

  boot() {
    return {
      status: "online",
      aesthetics: this.purple ? "purple.noir" : "default",
      mode: this.mode,
      timestamp: new Date().toISOString()
    };
  }
}

const nyx = new System({ mode: "focused", purple: true });
nyx.boot();
